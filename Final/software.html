private void countDownToLaunch() {

bol.setFilled(true);
bol.setFillColor(Color.RED);
add(bol, x, y);
for (int i = 50; i > 0; i--) {
bol.setVisible(false);
pause(i * 5);
bol.setVisible(true);
pause(i * 5);
if (i % 10 == 0) {
println("Countdown: t-minus: " + i / 10);
}
}
bol.setVisible(false);
}
